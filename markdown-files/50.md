#include <iostream>

class B {

class D2 : virtual public B {

    int get_i() { return B::get_i() + j2; }

class MM : public D, public D2 {

    int get_i() { return x; }
};

int main() {
    B *o = new MM();
    cout << o->get_i() << "\n";
    MM *p = dynamic_cast<MM*>(o);
    if (p) cout << p->get_i() << "\n";
    D *p2 = dynamic_cast<D*>(o);
    if (p2) cout << p2->get_i() << "\n";
    return 0;
}
